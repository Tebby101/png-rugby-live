async function fetchMatches() { const listContainer = document.getElementById('match-list'); try { const today = new Date().toISOString().split('T')[0]; const url = `https://www.thesportsdb.com/api/v1/json/3/eventsday.php?d=${today}&s=Rugby%20League`; const response = await fetch(url); const data = await response.json(); if (!data.events) { listContainer.innerHTML = '

No rugby matches scheduled for today.

'; return; } listContainer.innerHTML = ''; data.events.forEach(event => { const matchCard = document.createElement('div'); matchCard.className = 'match-card'; matchCard.innerHTML = `
${event.strTime || 'TBA'}

${event.strEvent}

${event.strLeague}

Watch Stream
`; listContainer.appendChild(matchCard); }); } catch (error) { console.error(error); listContainer.innerHTML = '

Failed to load matches.

'; } }
